Modelo do Banco de Dados

5 tabelas com relacionamentos via chaves estrangeiras

categories
PK idINTEGER
nameVARCHAR(100)
slugVARCHAR(100)
created_atTIMESTAMP
products
PK idINTEGER
nameVARCHAR(200)
priceDECIMAL
FK category_idINTEGER
brandVARCHAR(100)
sizesVARCHAR(100)
stockINTEGER
customers
PK idINTEGER
nameVARCHAR(200)
emailVARCHAR(200)
phoneVARCHAR(20)
addressTEXT
orders
PK idINTEGER
FK customer_idINTEGER
statusVARCHAR(20)
totalDECIMAL
shipping_addressTEXT
order_items
PK idINTEGER
FK order_idINTEGER
FK product_idINTEGER
sizeVARCHAR(5)
quantityINTEGER
unit_priceDECIMAL
subtotalDECIMAL
categories 1 → N products customers 1 → N orders orders 1 → N order_items products 1 → N order_items
